    import { existsSync, readdirSync } from 'node:fs';
    import { isAbsolute, join, resolve } from 'node:path';
    import { createJiti } from 'jiti';
    import type { ResolvedConfig } from '../config/types.js';
    import type { EntityMetadata } from '../user-entities/index.js';
    import { discoverEntityFiles, extractEntityMetadata } from '../user-entities/index.js';
    import type { EntitySnapshotFile } from '../snapshot/types.js';
    import { entitySnapshotPath, snapshotPaths, REMOVED_SUBDIR_NAME } from '../snapshot/paths.js';
    import { readJournal, readEntitySnapshot } from '../snapshot/read.js';
    import { loadMigrationFile } from '../runner/load-migration-module.js';
    import type { ValidateContext, MigrationRecord } from './types.js';

    export interface BuildValidateContextArgs {
      cwd: string;
      config: ResolvedConfig;
    }

    /**
     * Single-pass loader. Reads ALL data sources before returning so rules
     * are pure transforms with no I/O. Research §Pattern 2.
     *
     * jiti config matrix:
     *  - Entity files: per-call createJiti({tryNative:false, fsCache:false,
     *    moduleCache:false}) — copies the verbatim Pitfall-1 block from
     *    src/user-entities/load.ts so an edit between calls is observed.
     *  - Migration files: loadMigrationFile (uses {tryNative:true}) — correct
     *    for CI-static migration files.
     *
     * Pitfall 7: migration.reads is ReadonlyArray<AnyElectroEntity>; we
     * extract entity NAMES via (e as {model:{entity:string}}).model.entity
     * and store as readonly string[].
     */
    export async function buildValidateContext(args: BuildValidateContextArgs): Promise<ValidateContext> {
      const { cwd, config } = args;
      const paths = snapshotPaths(cwd);

      // 1. Live entities — per-call jiti for Pitfall 1.
      const entityFiles = await discoverEntityFiles({ cwd, entitiesConfig: config.entities });
      const liveEntities = new Map<string, EntityMetadata>();
      for (const f of entityFiles) {
        const jiti = createJiti(import.meta.url, { tryNative: false, fsCache: false, moduleCache: false });
        const mod = (await jiti.import(f)) as Record<string, unknown>;
        for (const m of extractEntityMetadata(mod, f)) liveEntities.set(m.entityName, m);
      }

      // 2. Journaled entity names.
      const journaledEntityNames = new Set<string>();
      if (existsSync(paths.journal)) {
        const journal = readJournal(paths.journal);
        for (const e of journal.entries) journaledEntityNames.add(e.entity);
      }

      // 3. Snapshots — read every <Entity>.snapshot.json under snapshotsDir.
      const snapshots = new Map<string, EntitySnapshotFile>();
      if (existsSync(paths.snapshotsDir)) {
        for (const f of readdirSync(paths.snapshotsDir)) {
          if (!f.endsWith('.snapshot.json')) continue;
          const entityName = f.slice(0, -'.snapshot.json'.length);
          const path = entitySnapshotPath(cwd, entityName);
          snapshots.set(entityName, readEntitySnapshot(path));
        }
      }

      // 4. Migrations — discover folders, load each via loadMigrationFile,
      //    extract MigrationRecord with reads names (Pitfall 7).
      const migrationsDirAbs = isAbsolute(config.migrations) ? config.migrations : resolve(cwd, config.migrations);
      const migrations: MigrationRecord[] = [];
      if (existsSync(migrationsDirAbs)) {
        for (const folderName of readdirSync(migrationsDirAbs)) {
          const folder = join(migrationsDirAbs, folderName);
          const migrationFile = join(folder, 'migration.ts');
          if (!existsSync(migrationFile)) continue;
          const m = await loadMigrationFile(migrationFile);
          // Pitfall 7: extract reads names.
          const reads: string[] = (m.reads ?? []).map((e) => (e as { model: { entity: string } }).model.entity);
          migrations.push({
            id: folderName,
            entityName: (m.from as { model: { entity: string } }).model.entity,
            fromVersion: String((m.from as { model: { version: string | number } }).model.version),
            toVersion: String((m.to as { model: { version: string | number } }).model.version),
            reads,
            folderPath: folder,
          });
        }
      }
      migrations.sort((a, b) =>
        a.entityName !== b.entityName
          ? a.entityName.localeCompare(b.entityName)
          : Number(a.fromVersion) - Number(b.fromVersion),
      );

      // 5. Acknowledged removals — read .electrodb-migrations/removed/*.tombstone.
      const removedDir = join(paths.root, REMOVED_SUBDIR_NAME);
      const acknowledgedRemovals = new Set<string>();
      if (existsSync(removedDir)) {
        for (const f of readdirSync(removedDir)) {
          if (f.endsWith('.tombstone')) acknowledgedRemovals.add(f.slice(0, -'.tombstone'.length));
        }
