    export interface ScaffoldPerfFixtureOpts {
      rootDir: string;             // absolute, fresh mkdtempSync directory

    /**
     * Generates a complete validate-able skeleton: entities + snapshots + journal +

      const { rootDir, entityCount = 20, migrationsPerEntity = 2 } = opts;

      // 1. Entity directory
      const entitiesDir = join(rootDir, 'src/database/entities');
      mkdirSync(entitiesDir, { recursive: true });
      const migrationsDir = join(rootDir, 'src/database/migrations');
      mkdirSync(migrationsDir, { recursive: true });
      const snapshotsDir = join(rootDir, '.electrodb-migrations/snapshots');
      mkdirSync(snapshotsDir, { recursive: true });

      const journalEntries: { entity: string; snapshot: string }[] = [];

          entity: entityName,
          service: 'app',
          attributes: {
            id: { type: 'string', required: true, hidden: false, readOnly: false, field: 'id' },
            name: { type: 'string', required: true, hidden: false, readOnly: false, field: 'name' },
            createdAt: { type: 'string', required: false, hidden: false, readOnly: false, field: 'createdAt' },
          },
          indexes: {
            primary: {
              type: 'primary',
              pk: { field: 'pk', composite: ['id'] },
              sk: { field: 'sk', composite: [] },
            },
          },
        };

        // 2. Live entity TS file (uses '2' as model.version to align with 2 migrations per entity).
        const entityTs = renderFrozenEntitySource({ projection, version: String(migrationsPerEntity + 1) });
        writeFileSync(join(entitiesDir, `${entityName}.ts`), entityTs.replace('Entity', 'Entity'), 'utf8');

        // 3. Per-entity migrations + frozenSnapshots
        const frozenSnapshots: { migrationId: string; v1Sha256: string; v2Sha256: string }[] = [];
        for (let m = 1; m <= migrationsPerEntity; m++) {

          writeFileSync(join(folder, 'v2.ts'), v2Source, 'utf8');

            join(folder, 'migration.ts'),

            migrationId,
            v1Sha256: computeIntegrityHash(v1Source),
            v2Sha256: computeIntegrityHash(v2Source),
          });
        }

        // 4. Snapshot file (fingerprint must match what fingerprintEntityModel produces on the entity — use a placeholder hash; the perf test does not assert VAL-01 cleanliness, only wall-time).
        // For SC-1 a clean repo is required — the most reliable way is to import fingerprintEntityModel here and project the live entity. For the budget test specifically, a clean fixture is required. Use a follow-up runtime step in the test itself to fingerprint and write the matching snapshot. See Task 2 below.
        const snapshotPath = join(snapshotsDir, `${entityName}.snapshot.json`);
        writeFileSync(

          JSON.stringify(
            {
              schemaVersion: FRAMEWORK_SNAPSHOT_VERSION,
              fingerprint: 'sha256:PLACEHOLDER_REPLACE_BY_TEST_RUNTIME',
              projection,
              frozenSnapshots,
            },
            null,
            2,
          ),
          'utf8',
        );
        journalEntries.push({ entity: entityName, snapshot: `snapshots/${entityName}.snapshot.json` });
      }

      // 5. Journal
      writeFileSync(
        join(rootDir, '.electrodb-migrations/_journal.json'),
        JSON.stringify(
          {
            schemaVersion: FRAMEWORK_SNAPSHOT_VERSION,
            minSchemaVersion: FRAMEWORK_SNAPSHOT_VERSION,
            entries: journalEntries,
          },
          null,
          2,
        ),
        'utf8',
      );
    }

    Pseudocode:
    ```typescript
    import { mkdtempSync, readFileSync, readdirSync, writeFileSync } from 'node:fs';
    import { tmpdir } from 'node:os';
    import { join } from 'node:path';
    import { performance } from 'node:perf_hooks';
    import { describe, expect, it } from 'vitest';
    import { createJiti } from 'jiti';
    import { fingerprintEntityModel } from '../../../src/safety/fingerprint-projection.js';
    import { buildValidateContext, runAllRules } from '../../../src/validate/index.js';
    import { scaffoldPerfFixture } from '../../fixtures/validate/_shared/scaffold-perf-fixture.js';

    describe('SC-1 perf gate — validate < 1000ms on 20-entity / 40-migration fixture', () => {
      it('exits zero in < 1000ms on a clean repository', async () => {
        const root = mkdtempSync(join(tmpdir(), 'edbm-perf-'));
        scaffoldPerfFixture({ rootDir: root });

        // Fix snapshot fingerprints to match live entities (sanity step — see Task 1 NOTE).
        const entitiesDir = join(root, 'src/database/entities');
        const snapsDir = join(root, '.electrodb-migrations/snapshots');
        const jiti = createJiti(import.meta.url, { tryNative: false, fsCache: false, moduleCache: false });
        for (const f of readdirSync(entitiesDir)) {
          if (!f.endsWith('.ts')) continue;
          const mod = (await jiti.import(join(entitiesDir, f))) as Record<string, unknown>;
          // Find the entity export
          for (const [k, v] of Object.entries(mod)) {
            if (typeof v !== 'object' || v === null) continue;
            const inst = v as { model?: unknown };
            if (typeof inst.model !== 'object' || inst.model === null) continue;
            const fp = fingerprintEntityModel(inst.model);
            const snapPath = join(snapsDir, `${k}.snapshot.json`);
            const snap = JSON.parse(readFileSync(snapPath, 'utf8'));
            snap.fingerprint = `sha256:${fp.fingerprint}`;
            snap.projection = fp.projection;
            writeFileSync(snapPath, JSON.stringify(snap, null, 2), 'utf8');
          }
        }

        const config = {
          entities: 'src/database/entities',
          migrations: 'src/database/migrations',
          migrationStartVersions: {},
        } as never;

        const start = performance.now();
        const ctx = await buildValidateContext({ cwd: root, config });
        const findings = runAllRules(ctx);
        const elapsed = performance.now() - start;

        expect(findings).toHaveLength(0);
        if (elapsed > 700) {
          // eslint-disable-next-line no-console
          console.warn(`SC-1 early-warning: validate elapsed ${elapsed.toFixed(0)}ms (budget 1000ms; warning threshold 700ms).`);
        }
        expect(elapsed, `SC-1 budget breached. Per RESEARCH §OQ7, enable jiti({fsCache:true}) for the validate path (config.validate.fsCache — v0.2 enhancement). Elapsed: ${elapsed.toFixed(0)}ms.`).toBeLessThan(1000);
      }, 30_000);   // 30s vitest timeout — generous for slow CI runners
    });
    ```

    Pitfall to avoid (Pitfall 1 + OQ7): the fingerprint-rewrite step uses a per-call jiti instance with caches disabled — same convention as buildValidateContext itself. Reusing a long-lived jiti across the 20 entities here would TIME-DISTORT the perf result.

    Run: `pnpm vitest run tests/unit/validate/performance.test.ts` → MUST PASS within 1000ms on developer hardware. If it fails on CI, the failure message tells the operator how to fix it.
