    export interface RunValidateArgs {
      cwd: string;
      configFlag?: string;
      json?: boolean;
    }

    /**
     * VAL-09 — `validate`. Pure FS rules engine. Loads ValidateContext via
     * Wave 1's buildValidateContext, runs the 8-rule registry, prints
     * findings as a table (default) or JSON (--json), then exits 0 (clean)
     * or 1 (findings).
     *
     * CLI-02: this command MUST NOT import @aws-sdk/client-dynamodb. The
     * Wave 0 file-only-commands grep guard catches regressions.
     */
    export async function runValidate(args: RunValidateArgs): Promise<void> {
      const { config, cwd } = await resolveCliConfig({
        cwd: args.cwd,
        ...(args.configFlag !== undefined ? { configFlag: args.configFlag } : {}),
      });
      const ctx = await buildValidateContext({ cwd, config });
      const findings = runAllRules(ctx);

      if (args.json === true) {
        process.stdout.write(`${JSON.stringify(findings, null, 2)}\n`);
      } else if (findings.length === 0) {
        log.ok('validate: no findings.');
      } else {
        // Human-readable table.
        const table = createTable({
          head: [c.bold('rule'), c.bold('entity'), c.bold('migration'), c.bold('message'), c.bold('fix')],
        });
        for (const f of findings) {
          table.push([
            c.warn(f.rule),
            f.entityName ?? '-',
            f.migrationId ?? '-',
            f.message,
            c.dim(f.fix),
          ]);
        }
        process.stderr.write(`${table.toString()}\n`);
        log.err(`validate: ${findings.length} finding${findings.length === 1 ? '' : 's'} blocking merge.`);
      }

      process.exit(findings.length > 0 ? EXIT_CODES.USER_ERROR : EXIT_CODES.OK);
    }

    export function registerValidateCommand(program: Command): void {
      program
        .command('validate')
        .description('Run the 8-rule CI gate; exits non-zero on findings (VAL-01..09)')
        .option('--json', 'Emit findings as JSON for CI consumption (OQ8)', false)
        .action(async (opts: { json?: boolean }) => {
          try {
            const configFlag = program.opts<{ config?: string }>().config;
            await runValidate({
              cwd: process.cwd(),
              ...(configFlag !== undefined ? { configFlag } : {}),
              json: opts.json ?? false,
            });
          } catch (err) {
            const message = err instanceof Error ? err.message : String(err);
            const remediation = (err as { remediation?: string }).remediation;
            log.err(message, remediation);
            process.exit(EXIT_CODES.USER_ERROR);
          }
        });
    }

    import type { Command } from 'commander';
    import { EXIT_CODES } from '../output/exit-codes.js';
    import { log } from '../output/log.js';
    import { resolveCliConfig } from '../shared/resolve-config.js';
    import { buildValidateContext } from '../../validate/index.js';
    import { scaffoldAcknowledgeRemoval } from '../../scaffold/index.js';

    export interface RunAcknowledgeRemovalArgs {
      cwd: string;
      configFlag?: string;
      entityName: string;
    }

    /**
     * VAL-10 — `acknowledge-removal <entity>`. Writes a zero-byte tombstone
     * after journal + live-entity guards. Pure FS — no DDB.
     *
     * Implementation notes:
     *   - Reuses buildValidateContext to populate journaledEntityNames +
     *     liveEntities; these are the only two pieces scaffoldAcknowledgeRemoval
     *     needs. The full context load is acceptable here because the cost
     *     is O(entities + migrations) and runs once per CLI invocation (no
     *     hot path).
     */
    export async function runAcknowledgeRemoval(args: RunAcknowledgeRemovalArgs): Promise<void> {
      const { config, cwd } = await resolveCliConfig({
        cwd: args.cwd,
        ...(args.configFlag !== undefined ? { configFlag: args.configFlag } : {}),
      });
      const ctx = await buildValidateContext({ cwd, config });
      const liveEntityNames = new Set([...ctx.liveEntities.keys()]);
      await scaffoldAcknowledgeRemoval({
        cwd,
        entityName: args.entityName,
        journaledEntityNames: ctx.journaledEntityNames,
        liveEntityNames,
      });
      log.ok(`Entity '${args.entityName}' recorded as intentionally removed. Subsequent \`validate\` will exit zero for this entity.`);
    }

    export function registerAcknowledgeRemovalCommand(program: Command): void {
      program
        .command('acknowledge-removal <entity>')
        .description('Record an entity as intentionally removed; subsequent validate exits zero (VAL-10)')
        .action(async (entity: string) => {
          try {
            const configFlag = program.opts<{ config?: string }>().config;
            await runAcknowledgeRemoval({
              cwd: process.cwd(),
              ...(configFlag !== undefined ? { configFlag } : {}),
              entityName: entity,
            });
          } catch (err) {
            const code = (err as { code?: string }).code;
            const message = err instanceof Error ? err.message : String(err);
            if (code === 'EDB_ACKNOWLEDGE_REMOVAL_ERROR') {
              log.err(message);
              process.exit(EXIT_CODES.USER_ERROR);
            }
            const remediation = (err as { remediation?: string }).remediation;
            log.err(message, remediation);
            process.exit(EXIT_CODES.USER_ERROR);
          }
        });
    }
