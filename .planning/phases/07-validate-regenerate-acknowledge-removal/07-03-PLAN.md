    /**
     * VAL-01 — drift-without-migration: an entity's current shape differs
     * from its latest snapshot AND no scaffolded migration covers the diff.
     * Pure function; no I/O. Consumes ValidateContext only.
     *
     * Logic:
     *   For each entity in ctx.liveEntities:
     *     - Compute live fingerprint via fingerprintEntityModel(entity.model).
     *     - Read snapshot.fingerprint (already 'sha256:<hex>').
     *     - If equal -> no drift -> skip.
     *     - If different -> drift exists. Check whether ctx.migrations
     *       contains a covering migration for this entity, where covering
     *       means an existing migration with fromVersion equal to the
     *       entity's CURRENT modelVersion (i.e. the snapshot baseline).
     *       That migration represents an in-flight scaffold — its toVersion
     *       extends the model. If found, drift is covered; skip.
     *     - Otherwise emit finding.
     *
     * Fix: 'run `electrodb-migrations create --entity <name> --name <slug>`'
     */
    import { fingerprintEntityModel } from '../../safety/fingerprint-projection.js';
    import type { ValidateContext, ValidateRule, ValidationFinding } from '../types.js';

    export const driftWithoutMigrationRule: ValidateRule = {
      name: 'drift-without-migration',
      check(ctx: ValidateContext): ValidationFinding[] {
        const findings: ValidationFinding[] = [];
        for (const [entityName, meta] of ctx.liveEntities) {
          const snapshot = ctx.snapshots.get(entityName);
          if (snapshot === undefined) continue;   // greenfield — no prior snapshot to drift from; baseline handles this
          const live = fingerprintEntityModel((meta.entityInstance as { model: unknown }).model);
          const liveFingerprint = `sha256:${live.fingerprint}`;
          if (liveFingerprint === snapshot.fingerprint) continue;
          // Drift exists — check for covering migration.
          const baselineVersion = String(meta.modelVersion);
          const hasCovering = ctx.migrations.some(
            (m) => m.entityName === entityName && m.fromVersion === baselineVersion,
          );
          if (hasCovering) continue;
          findings.push({
            rule: 'drift-without-migration',
            severity: 'error',
            entityName,
            message: `Entity '${entityName}' (${meta.sourceFilePath}) has shape drift but no scaffolded migration covers it.`,
            fix: `run \`electrodb-migrations create --entity ${entityName} --name <slug>\``,
            detail: { entityName, sourceFilePath: meta.sourceFilePath, baselineVersion },
          });
        }
        return findings;
      },
    };
    ```

    /**
     * VAL-02 — version-skew: entity's model.version must equal the highest
     * scaffolded toVersion. Pure function.
     *
     * Logic:
     *   For each (entityName, meta) in ctx.liveEntities:
     *     - Filter ctx.migrations to the entity; collect Number(m.toVersion).
     *     - If list is empty: skip (entity with no migrations has implicit
     *       toVersion === modelVersion; no skew possible).
     *     - maxToVersion = Math.max(...list).
     *     - If maxToVersion !== Number(meta.modelVersion) -> emit finding.
     *
     * Fix: 'update model.version to <maxToVersion> in your entity source'
     */
    import type { ValidateContext, ValidateRule, ValidationFinding } from '../types.js';

    export const versionSkewRule: ValidateRule = {
      name: 'version-skew',
      check(ctx: ValidateContext): ValidationFinding[] {
        const findings: ValidationFinding[] = [];
        for (const [entityName, meta] of ctx.liveEntities) {
          const toVersions = ctx.migrations
            .filter((m) => m.entityName === entityName)
            .map((m) => Number(m.toVersion));
          if (toVersions.length === 0) continue;
          const maxTo = Math.max(...toVersions);
          const actual = Number(meta.modelVersion);
          if (maxTo === actual) continue;
          findings.push({
            rule: 'version-skew',
            severity: 'error',
            entityName,
            message: `Entity '${entityName}' (${meta.sourceFilePath}) declares model.version='${meta.modelVersion}' but the latest scaffolded migration's toVersion is '${maxTo}'.`,
            fix: `update model.version to '${maxTo}' in ${meta.sourceFilePath}`,
            detail: { entityName, expected: maxTo, actual: meta.modelVersion, sourceFilePath: meta.sourceFilePath },
          });
        }
        return findings;
      },
    };
    ```

    /**
     * VAL-03 — sequence-gaps: per-entity migration sequence must be
     * consecutive starting at 1 (or at config.migrationStartVersions[entity].version).
     * Pure function. Pitfall 3: sort by Number(fromVersion); never by id.
     * Pitfall 4: respect migrationStartVersions.
     *
     * Logic:
     *   For each entityName seen in ctx.migrations:
     *     - sorted = migrations.filter(entityName).sort((a,b) => Number(a.fromVersion) - Number(b.fromVersion))
     *     - expected = ctx.config.migrationStartVersions[entityName]?.version ?? 1
     *     - For each m in sorted:
     *         if Number(m.fromVersion) !== expected -> emit finding (gap)
     *         expected = Number(m.toVersion)
     *
     * Fix: 'create the missing migration for v<expected>->v<missing>'
     */
    import type { ValidateContext, ValidateRule, ValidationFinding } from '../types.js';

    export const sequenceGapsRule: ValidateRule = {
      name: 'sequence-gaps',
      check(ctx: ValidateContext): ValidationFinding[] {
        const findings: ValidationFinding[] = [];
        const byEntity = new Map<string, typeof ctx.migrations[number][]>();
        for (const m of ctx.migrations) {
          const list = byEntity.get(m.entityName) ?? [];
          list.push(m);
          byEntity.set(m.entityName, list);
        }
        for (const [entityName, list] of byEntity) {
          const sorted = [...list].sort((a, b) => Number(a.fromVersion) - Number(b.fromVersion));
          let expected = ctx.config.migrationStartVersions[entityName]?.version ?? 1;
          for (const m of sorted) {
            const from = Number(m.fromVersion);
            if (from !== expected) {
              findings.push({
                rule: 'sequence-gaps',
                severity: 'error',
                entityName,
                migrationId: m.id,
                message: `Entity '${entityName}' has a sequence gap: expected migration starting at v${expected}, but migration '${m.id}' starts at v${from}.`,
                fix: `create the missing migration for v${expected} -> v${from} (or fix the fromVersion of '${m.id}')`,
                detail: { entityName, expected, actualFromVersion: from, migrationId: m.id },
              });
              // Continue iterating so additional gaps in the same entity are reported.
              expected = Number(m.toVersion);
              continue;
            }
            expected = Number(m.toVersion);
          }
        }
        return findings;
      },
    };
    ```

    Pitfall to avoid (Pitfall 3): never sort by `m.id` (timestamp prefix is unrelated to fromVersion). Pitfall 4: ALWAYS read `ctx.config.migrationStartVersions[entityName]?.version ?? 1` — never hardcode `1`.
  </behavior>

    /**
     * VAL-04 — parallel-branch-collision: two or more migrations claim the
     * same (entityName, fromVersion). Pure function. Pitfall 3: never sort
     * by migration ID; group by (entityName, fromVersion) only.
     *
     * Logic:
     *   key = `${entityName}::${fromVersion}`
     *   group migrations by key
     *   for each group with length > 1, emit ONE finding listing ALL
     *     colliding migration ids (sorted asc for deterministic output).
     *
     * Fix: 'rebase the later branch and run `electrodb-migrations create --regenerate <id>`'
     */
    import type { ValidateContext, ValidateRule, ValidationFinding } from '../types.js';

    export const parallelBranchCollisionRule: ValidateRule = {
      name: 'parallel-branch-collision',
      check(ctx: ValidateContext): ValidationFinding[] {
        const findings: ValidationFinding[] = [];
        const groups = new Map<string, string[]>();   // key -> [migrationIds]
        const meta = new Map<string, { entityName: string; fromVersion: string }>();
        for (const m of ctx.migrations) {
          const key = `${m.entityName}::${m.fromVersion}`;
          const ids = groups.get(key) ?? [];
          ids.push(m.id);
          groups.set(key, ids);
          meta.set(key, { entityName: m.entityName, fromVersion: m.fromVersion });
        }
        for (const [key, ids] of groups) {
          if (ids.length < 2) continue;
          const sortedIds = [...ids].sort();
          const m = meta.get(key);
          if (!m) continue;
          findings.push({
            rule: 'parallel-branch-collision',
            severity: 'error',
            entityName: m.entityName,
            message: `Entity '${m.entityName}' has ${ids.length} migrations claiming fromVersion='${m.fromVersion}': ${sortedIds.join(', ')}.`,
            fix: `rebase the later branch onto main and run \`electrodb-migrations create --regenerate <id>\``,
            detail: { entityName: m.entityName, fromVersion: m.fromVersion, collidingIds: sortedIds },
          });
        }
        return findings;
      },
    };
    ```

    Pitfall to avoid (Pitfall 3): if you find yourself sorting by migration ID for the gap detection logic of THIS rule, stop — VAL-04 has no sort dependency at all; grouping is all you need.
