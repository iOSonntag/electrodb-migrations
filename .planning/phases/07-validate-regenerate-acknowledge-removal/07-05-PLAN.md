    import { existsSync, readFileSync, writeFileSync } from 'node:fs';
    import { isAbsolute, join, resolve } from 'node:path';
    import { type Drift, classifyDrift } from '../drift/classify.js';
    import { EDBMigrationError } from '../errors/base.js';
    import { type EntityProjection, fingerprintEntityModel } from '../safety/fingerprint-projection.js';
    import { entitySnapshotPath } from '../snapshot/paths.js';
    import { readEntitySnapshot } from '../snapshot/read.js';
    import { writeEntitySnapshot } from '../snapshot/write.js';
    import { FRAMEWORK_SNAPSHOT_VERSION } from '../snapshot/version.js';
    import type { EntitySnapshotFile } from '../snapshot/types.js';
    import type { EntityMetadata } from '../user-entities/index.js';
    import { loadMigrationFile } from '../runner/load-migration-module.js';
    import { renderFrozenEntitySource } from './frozen-snapshot.js';
    import { computeIntegrityHash } from './integrity-hash.js';

    /**
     * SCF-08 — `create --regenerate <id>`. Re-emits v1.ts and v2.ts for an
     * existing migration after a rebase, preserves migration.ts byte-for-byte
     * (SC-3), and updates the frozenSnapshots entry by migrationId
     * (Pitfall 5 — never by array index).
     *
     * INVARIANT: migration.ts is never written to disk. The function reads its
     * bytes BEFORE any write and reads them AGAIN after all writes; if they
     * diverge, it throws (belt-and-suspenders defense for SC-3).
     */
    export class EDBRegenerateError extends EDBMigrationError {
      readonly code = 'EDB_REGENERATE_ERROR' as const;
    }

    export interface ScaffoldRegenerateArgs {
      cwd: string;
      migrationsDir: string;
      migrationId: string;
      liveEntityMeta: EntityMetadata | null;
      sourceFilePath: string | null;
    }

    export interface ScaffoldRegenerateResult {
      migrationId: string;
      migrationFolderPath: string;
      fromVersion: string;
      toVersion: string;
      v1Sha256: string;
      v2Sha256: string;
      drifts: ReadonlyArray<Drift>;
    }

    export async function scaffoldRegenerate(args: ScaffoldRegenerateArgs): Promise<ScaffoldRegenerateResult> {
      const { cwd, migrationsDir, migrationId, liveEntityMeta } = args;

      // 1. Locate the migration folder.
      const migrationsDirAbs = isAbsolute(migrationsDir) ? migrationsDir : resolve(cwd, migrationsDir);
      const migrationFolderPath = join(migrationsDirAbs, migrationId);
      const migrationFile = join(migrationFolderPath, 'migration.ts');
      const v1Path = join(migrationFolderPath, 'v1.ts');
      const v2Path = join(migrationFolderPath, 'v2.ts');

      if (!existsSync(migrationFolderPath) || !existsSync(migrationFile)) {
        throw new EDBRegenerateError(
          `Migration folder not found: ${migrationFolderPath}. Cannot regenerate '${migrationId}'.`,
          { migrationId, migrationFolderPath },
        );
      }

      // 2. Read existing migration to extract entityName + version pair.
      const existingMigration = await loadMigrationFile(migrationFile);
      const entityName = (existingMigration.from as { model: { entity: string } }).model.entity;

      // 3. OQ-R2 guard: live entity must exist.
      if (liveEntityMeta === null || liveEntityMeta.entityName !== entityName) {
        throw new EDBRegenerateError(
          `Entity '${entityName}' not found in entities directory. Did you mean to run \`acknowledge-removal ${entityName}\` instead?`,
          { entityName, migrationId },
        );
      }

      // 4. Read migration.ts BYTES (SC-3 belt-and-suspenders before).
      const migrationTsBefore = readFileSync(migrationFile);

      // 5. Read current snapshot — represents the new v1 baseline (rebased main).
      const snapshotPath = entitySnapshotPath(cwd, entityName);
      if (!existsSync(snapshotPath)) {
        throw new EDBRegenerateError(
          `Snapshot for entity '${entityName}' not found. Run \`baseline\` first.`,
          { entityName },
        );
      }
      const snapshot = readEntitySnapshot(snapshotPath);
      const baselineProjection = snapshot.projection as unknown as EntityProjection;

      // 6. Project live entity (= new v2 shape).
      const live = fingerprintEntityModel((liveEntityMeta.entityInstance as { model: unknown }).model);

      // 7. Derive new fromVersion / toVersion.
      const newFromVersion = String(liveEntityMeta.modelVersion);   // current entity model.version IS the new from
      const newFromNum = Number(newFromVersion);
      if (Number.isNaN(newFromNum)) {
        throw new EDBRegenerateError(
          `Cannot regenerate: '${entityName}'.model.version='${newFromVersion}' is not numeric.`,
          { entityName, newFromVersion },
        );
      }
      const newToVersion = String(newFromNum + 1);

      // 8. Compute drifts for the CLI's diff render.
      const drifts = classifyDrift(baselineProjection, live.projection);

      // 9. Re-emit v1.ts and v2.ts.
      const v1Source = renderFrozenEntitySource({ projection: baselineProjection, version: newFromVersion });
      const v2Source = renderFrozenEntitySource({ projection: live.projection, version: newToVersion });
      const v1Sha256 = computeIntegrityHash(v1Source);
      const v2Sha256 = computeIntegrityHash(v2Source);
      writeFileSync(v1Path, v1Source, 'utf8');
      writeFileSync(v2Path, v2Source, 'utf8');

      // 10. SC-3 belt-and-suspenders AFTER: re-read migration.ts and assert byte-equal.
      const migrationTsAfter = readFileSync(migrationFile);
      if (!migrationTsBefore.equals(migrationTsAfter)) {
        throw new EDBRegenerateError(
          `Internal error: migration.ts bytes changed during regenerate (this should never happen). Expected SC-3 invariant violation.`,
          { migrationId, migrationFile },
        );
      }

      // 11. Update frozenSnapshots — Pitfall 5: find-by-migrationId, never by index.
      const prevFrozen = snapshot.frozenSnapshots ?? [];
      const idx = prevFrozen.findIndex((e) => e.migrationId === migrationId);
      const updatedFrozen =
        idx >= 0
          ? prevFrozen.map((e, i) => (i === idx ? { migrationId, v1Sha256, v2Sha256 } : e))
          : [...prevFrozen, { migrationId, v1Sha256, v2Sha256 }];

      // 12. Write snapshot.
      const updatedSnapshot: EntitySnapshotFile = {
        schemaVersion: snapshot.schemaVersion ?? FRAMEWORK_SNAPSHOT_VERSION,
        fingerprint: `sha256:${live.fingerprint}`,
        projection: live.projection as unknown as Record<string, unknown>,
        frozenSnapshots: updatedFrozen,
      };
      writeEntitySnapshot(snapshotPath, updatedSnapshot);

      return {
        migrationId,
        migrationFolderPath,
        fromVersion: newFromVersion,
        toVersion: newToVersion,
        v1Sha256,
        v2Sha256,
        drifts,
      };
    }
    ```

    - OQ-R2: throw EDBRegenerateError with the named-error message when entity is removed.

    import { mkdirSync, writeFileSync } from 'node:fs';
    import { dirname } from 'node:path';
    import { EDBMigrationError } from '../errors/base.js';
    import { entityRemovedTombstonePath } from '../snapshot/paths.js';

    /**
     * VAL-10 — `acknowledge-removal <entity>`. Writes a zero-byte tombstone
     * file at .electrodb-migrations/removed/<EntityName>.tombstone after
     * three guard checks:
     *   1. entityName is non-empty + no path traversal (delegated to
     *      entityRemovedTombstonePath via Wave 0's V5 guard).
     *   2. entityName must NOT be a live entity — tombstoning a present
     *      entity is nonsensical.
     *   3. entityName MUST be in the journal — tombstoning an unknown
     *      entity is likely a typo (VAL-06 would never fire for it).
     *
     * Idempotent: if the tombstone already exists, overwrite with empty
     * content (no-op outcome).
     */
    export class EDBAcknowledgeRemovalError extends EDBMigrationError {
      readonly code = 'EDB_ACKNOWLEDGE_REMOVAL_ERROR' as const;
    }

    export interface ScaffoldAcknowledgeRemovalArgs {
      cwd: string;
      entityName: string;
      journaledEntityNames: ReadonlySet<string>;
      liveEntityNames: ReadonlySet<string>;
    }

    export async function scaffoldAcknowledgeRemoval(args: ScaffoldAcknowledgeRemovalArgs): Promise<void> {
      const { cwd, entityName, journaledEntityNames, liveEntityNames } = args;

      // Guard 1 (delegated): path-traversal validation occurs inside entityRemovedTombstonePath.
      // Resolve the path early so a bad entityName fails fast with the V5 guard's message.
      const tombstonePath = entityRemovedTombstonePath(cwd, entityName);

      // Guard 2: not a live entity.
      if (liveEntityNames.has(entityName)) {
        throw new EDBAcknowledgeRemovalError(
          `Entity '${entityName}' is currently defined in your entities directory. Remove it from the source first, then run \`acknowledge-removal\` to record the intentional removal.`,
          { entityName },
        );
      }

      // Guard 3: must be in journal.
      if (!journaledEntityNames.has(entityName)) {
        throw new EDBAcknowledgeRemovalError(
          `Entity '${entityName}' is not present in the snapshot journal. Either it was never baselined (typo?) or it has already been removed and tombstoned.`,
          { entityName },
        );
      }

      // Write zero-byte tombstone (idempotent).
      mkdirSync(dirname(tombstonePath), { recursive: true });
      writeFileSync(tombstonePath, '', 'utf8');
    }
    ```
