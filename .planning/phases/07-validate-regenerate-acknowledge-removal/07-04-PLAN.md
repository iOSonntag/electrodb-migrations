     * VAL-07 — reserved-namespace: user entity names must not start with '_'.
     * The framework reserves the underscore prefix for internal entities
     * (e.g. _migrations, _lock). Pure function; no I/O.
     */
    import type { ValidateContext, ValidateRule, ValidationFinding } from '../types.js';

    export const reservedNamespaceRule: ValidateRule = {
      name: 'reserved-namespace',
      check(ctx: ValidateContext): ValidationFinding[] {
        const findings: ValidationFinding[] = [];
        for (const [entityName, meta] of ctx.liveEntities) {
          if (!entityName.startsWith('_')) continue;

            rule: 'reserved-namespace',
            severity: 'error',
            entityName,
            message: `Entity '${entityName}' (${meta.sourceFilePath}) uses the reserved '_' prefix; that namespace is reserved for framework internal entities (_migrations, _lock).`,
            fix: `rename '${entityName}' to a name that does not start with '_' in ${meta.sourceFilePath}`,
            detail: { entityName, sourceFilePath: meta.sourceFilePath },
          });
        }
        return findings;
      },
    };

    /**
     * VAL-06 — removed-entities: an entity recorded in _journal.json that
     * is no longer present on disk AND has not been tombstoned via
     * `acknowledge-removal` is flagged. Pure function; no I/O (the data is
     * pre-loaded into ctx.journaledEntityNames + ctx.acknowledgedRemovals).
     *
     * Pitfall 6: the framework has no rename primitive in v0.1, so a rename
     * looks identical to a removal. The finding's message MUST include the
     * rename-hint so operators don't blindly tombstone a renamed entity.
     */
    import type { ValidateContext, ValidateRule, ValidationFinding } from '../types.js';

    export const removedEntitiesRule: ValidateRule = {
      name: 'removed-entities',
      check(ctx: ValidateContext): ValidationFinding[] {
        const findings: ValidationFinding[] = [];
        for (const entityName of ctx.journaledEntityNames) {
          if (ctx.liveEntities.has(entityName)) continue;
          if (ctx.acknowledgedRemovals.has(entityName)) continue;
          findings.push({
            rule: 'removed-entities',
            severity: 'error',
            entityName,
            message: `Entity '${entityName}' was previously snapshotted but is no longer found in the entities directory. If this is a removal, run \`acknowledge-removal\`. If this is a rename, restore the original name (or update the entity file to the new name) and run \`baseline\` after fixing.`,
            fix: `run \`electrodb-migrations acknowledge-removal ${entityName}\``,
            detail: { entityName },
          });
        }
        return findings;
      },
    };
    ```

    /**
     * VAL-05 — cross-entity ordering: for each migration M that declares
     * `reads: [Other]`, every Other-target must NOT have a later-sequenced
     * pending migration. Pure FS rule — operates on ctx.migrations.reads
     * (already extracted as string[] in Wave 1; Pitfall 7). Never queries
     * the _migrations DDB rows (Pitfall 2 — the apply-time check is
     * Phase 6's findBlockingReadsDependency; that is a DIFFERENT concern).
     *
     * Logic:
     *   For each m with m.reads.length > 0:
     *     For each target in m.reads:
     *       For each n in ctx.migrations:
     *         if n.entityName === target AND Number(n.fromVersion) >= Number(m.toVersion):
     *           emit finding naming m.id, n.id, target.
     */
    import type { ValidateContext, ValidateRule, ValidationFinding } from '../types.js';

    export const crossEntityOrderingRule: ValidateRule = {
      name: 'cross-entity-ordering',
      check(ctx: ValidateContext): ValidationFinding[] {
        const findings: ValidationFinding[] = [];
        for (const m of ctx.migrations) {
          if (m.reads.length === 0) continue;
          const mTo = Number(m.toVersion);
          for (const target of m.reads) {
            for (const n of ctx.migrations) {
              if (n.entityName !== target) continue;
              if (Number(n.fromVersion) < mTo) continue;
              findings.push({
                rule: 'cross-entity-ordering',
                severity: 'error',
                entityName: m.entityName,
                migrationId: m.id,
                message: `Migration '${m.id}' on '${m.entityName}' reads '${target}', but a later-sequenced pending migration '${n.id}' exists on '${target}' (fromVersion=${n.fromVersion}, M.toVersion=${m.toVersion}).`,
                fix: `apply or reorder: ensure '${n.id}' applies BEFORE '${m.id}', or remove '${target}' from '${m.id}'.reads if no longer needed`,
                detail: { migrationId: m.id, blockingMigrationId: n.id, readsTarget: target, mToVersion: m.toVersion, nFromVersion: n.fromVersion },
              });
            }
          }
        }
        return findings;
      },
    };
    ```

    Pitfalls to avoid:
    - Pitfall 2: never `import` from `@aws-sdk/*`; never query `_migrations` rows. The rule is FS-only.
    - Pitfall 7: `m.reads` is already `string[]` (Wave 1 extracted via `.model.entity`). Do NOT cast to `AnyElectroEntity[]` here.
    - Performance: the O(M*R*M) double-nested loop is fine for SC-1's ~40-migration fixture (worst case ~3200 iterations); a Map-based optimization is unnecessary at v0.1 scale.
  </behavior>

  <files>src/validate/rules/frozen-snapshot-edited.ts</files>
  <behavior>
    GREEN body:
    ```typescript
    /**
     * VAL-08 — frozen-snapshot-edited: for each frozenSnapshots entry on
     * each entity snapshot, recompute computeIntegrityHash on the actual
     * v1.ts and v2.ts files on disk and compare to the stored
     * v1Sha256/v2Sha256. Pitfall 5: lookup IS by migrationId, not by
     * array index. Wave 1's run-rules.ts places this rule LAST in the
     * registry because it touches the filesystem (~2*N file reads).
     *
     * Missing-folder / missing-file paths produce findings with a clear
     * message so operators distinguish deletion from edit.
     */
    import { existsSync, readFileSync } from 'node:fs';
    import { isAbsolute, join, resolve } from 'node:path';
    import { computeIntegrityHash } from '../../scaffold/integrity-hash.js';
    import type { ValidateContext, ValidateRule, ValidationFinding } from '../types.js';

    export const frozenSnapshotEditedRule: ValidateRule = {
      name: 'frozen-snapshot-edited',
      check(ctx: ValidateContext): ValidationFinding[] {
        const findings: ValidationFinding[] = [];
        const migrationsDirAbs = isAbsolute(ctx.config.migrations)
          ? ctx.config.migrations
          : resolve(ctx.cwd, ctx.config.migrations);
        for (const [entityName, snapshot] of ctx.snapshots) {
          for (const entry of snapshot.frozenSnapshots ?? []) {
            const folder = join(migrationsDirAbs, entry.migrationId);
            const v1Path = join(folder, 'v1.ts');
            const v2Path = join(folder, 'v2.ts');
            if (!existsSync(folder) || !existsSync(v1Path) || !existsSync(v2Path)) {
              findings.push({
                rule: 'frozen-snapshot-edited',
                severity: 'error',
                entityName,
                migrationId: entry.migrationId,
                message: `Frozen v1.ts/v2.ts for migration '${entry.migrationId}' (entity '${entityName}') not found at '${folder}'. The migration folder may have been deleted.`,
                fix: `restore the migration folder, or run \`electrodb-migrations create --regenerate ${entry.migrationId}\` to re-emit it`,
                detail: { entityName, migrationId: entry.migrationId, folderPath: folder },
              });
              continue;
            }
            const v1Hash = computeIntegrityHash(readFileSync(v1Path));
            const v2Hash = computeIntegrityHash(readFileSync(v2Path));
            const v1Match = v1Hash === entry.v1Sha256;
            const v2Match = v2Hash === entry.v2Sha256;
            if (v1Match && v2Match) continue;
            findings.push({
              rule: 'frozen-snapshot-edited',
              severity: 'error',
              entityName,
              migrationId: entry.migrationId,
              message: `Frozen schema files for migration '${entry.migrationId}' (entity '${entityName}') have been edited: ${[!v1Match ? 'v1.ts' : null, !v2Match ? 'v2.ts' : null].filter(Boolean).join(' + ')} hash mismatch.`,
              fix: `run \`electrodb-migrations create --regenerate ${entry.migrationId}\` to restore the canonical frozen sources, or revert your edits manually`,
              detail: {
                entityName,
                migrationId: entry.migrationId,
                v1Match,
                v2Match,
                v1Expected: entry.v1Sha256,
                v1Actual: v1Hash,
                v2Expected: entry.v2Sha256,
                v2Actual: v2Hash,
              },
            });
          }
        }
        return findings;
      },
    };
    ```

    Pitfalls to avoid:
    - Pitfall 5: `frozenSnapshots` is an Array — but lookup IS by `entry.migrationId` (the iteration is by entry, so this is intrinsic). The corollary is: never use `frozenSnapshots[i]` indexing in any consumer.
    - Performance: SC-1 < 1s permits ~40 migrations × 2 file reads = 80 syscalls + 80 SHA-256 hashes — well under budget on any CI runner.
  </behavior>
