    //
    // Rationale (RESEARCH §OQ7, README §6.6.2):
    //   When migration M reads entity Y via ctx.entity(Y), the on-disk shape of Y
    //   must match the source Y the migration imported. That holds iff every
    //   pending migration on Y is sequenced BEFORE M. If Y has any migration
    //   sequenced after M, on-disk Y is at a newer shape — rolling back M would
    //   re-introduce v1 records of M's entity that reference a now-stale Y shape.
    //
    //   The head-only rule (Step 5) catches within-entity ordering. CTX-08 extends
    //   the rule across entities: a Team migration sequenced after the User
    //   migration M, where M declares reads:[Team], blocks M's rollback. The user
    //   must roll back the Team migration first.
    //
    // Sequence comparison uses `fromVersion` (numeric string) per RESEARCH §A3 /
    // Pitfall 6 — `appliedAt` ISO timestamps are clock-skew-vulnerable in
    // multi-developer environments. `fromVersion` is sequence-monotonic per entity.
    if (targetRow.reads !== undefined && targetRow.reads.size > 0) {
      const blocking = findBlockingReadsDependency(allRows, targetRow);
      if (blocking !== undefined) {
        const err = new EDBRollbackNotPossibleError(
          `Cannot rollback ${args.migration.id}: migration ${blocking.id} on reads-target ` +
            `entity '${blocking.entityName}' has been applied since ${args.migration.id}. ` +
            `Roll back ${blocking.id} first.`,

    ```typescript
    /**
     * CTX-08 helper — finds a `_migrations` row that blocks rollback of
     * `targetRow` due to a cross-entity reads dependency.
     *
     * A row R is blocking IFF:
     *   1. R.entityName ∈ targetRow.reads (the row migrates a reads-target entity)
     *   2. R.status ∈ {'applied', 'finalized'} (it's an active dependency, NOT
     *      reverted or pending)
     *   3. parseInt(R.fromVersion) >= parseInt(targetRow.toVersion) (R moves the
     *      reads-target entity from a version >= the version M was authored against)
     *
     * The `>=` semantics: `targetRow.toVersion` is the version that M migrated TO
     * for its OWN entity. A reads-target migration with `fromVersion` equal to or
     * greater than that means the reads-target entity is now at a shape AT LEAST
     * one version newer than what M's source code expected. Rolling back M would
     * re-introduce v1 records of M's entity that reference a stale reads-target
     * shape.
     *
     * Returns the FIRST blocking row found (sorted by fromVersion ascending so
     * the earliest blocker is reported — the user fixes it first then re-runs).
     * Returns `undefined` if no blocking row exists.
     *
     * RESEARCH §A3 (lines 771): fromVersion comparison is clock-skew safe (vs
     * `appliedAt` ISO timestamps which can drift between machines).
     *
     * Cross-entity comparison caveat: R.fromVersion is the reads-target entity's
     * sequence; targetRow.toVersion is M's own entity's sequence. This comparison
     * is meaningful because each entity's version numbers are monotonically
     * increasing integer strings starting at '1' (the framework's authoring
     * convention — `model.version` bumps via ts-morph in Phase 2 always
     * increment). If a user opts an entity into a non-1 starting version via
     * `migrationStartVersions` (a Phase 7 concept), this comparison may produce
     * a false negative; Phase 7's `validate` gate is the authoritative ordering
     * check for those cases.
     */
    function findBlockingReadsDependency(
      allRows: MigrationsRow[],
      targetRow: MigrationsRow,
    ): MigrationsRow | undefined {
      if (targetRow.reads === undefined || targetRow.reads.size === 0) {
        return undefined;
      }
      const targetToVersionInt = Number.parseInt(targetRow.toVersion, 10);
      // Filter, then sort by fromVersion ascending so the earliest blocker is reported.
      const blockers = allRows
        .filter((r) => {
          if (!targetRow.reads!.has(r.entityName)) return false;
          if (r.status !== 'applied' && r.status !== 'finalized') return false;
          const rFromInt = Number.parseInt(r.fromVersion, 10);
          // NaN check: malformed fromVersion → treat as non-blocking (defensive — but log a warning later if v0.1 starts seeing these).
          if (Number.isNaN(rFromInt) || Number.isNaN(targetToVersionInt)) return false;
          return rFromInt >= targetToVersionInt;
        })
        .sort((a, b) => Number.parseInt(a.fromVersion, 10) - Number.parseInt(b.fromVersion, 10));
      return blockers[0];
    }
    ```
